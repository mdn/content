{{APIRef("HTML Drag and Drop API")}}

The read-only DataTransferItem.kind property returns a {{domxref("DataTransferItem")}} representing the drag data item kind: some text or some file.

Syntax

var itemKind = DataTransferItem.kind;

Return value

A {{domxref("DOMString")}} representing the drag data item's kind. It must be one of the following values:

'file'
If the drag data item is a file.
'string'
If the kind of drag data item is a plain Unicode string.

Example

This example shows the use of the kind property.

function drop_handler(ev) {
 console.log("Drop");
 ev.preventDefault();
 var data = event.dataTransfer.items;
 for (var i = 0; i < data.length; i += 1) {
   if ((data[i].kind == 'string') &&
       (data[i].type.match('^text/plain'))) {
     // This item is the target node
     data[i].getAsString(function (s){
       ev.target.appendChild(document.getElementById(s));
     });
   } else if ((data[i].kind == 'string') &&
              (data[i].type.match('^text/html'))) {
     // Drag data item is HTML
     console.log("... Drop: HTML");
   } else if ((data[i].kind == 'file') &&
              (data[i].type.match('^image/'))) {
     // Drag data item is an image file
     var f = data[i].getAsFile();
     console.log("... Drop: File ");
   }
 }
}
